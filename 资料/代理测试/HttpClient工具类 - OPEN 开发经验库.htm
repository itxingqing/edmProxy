057     * 发送HTTP_GET请求
058     * @see 该方法会自动关闭连接,释放资源
059     * @param requestURL    请求地址(含参数)
060     * @param decodeCharset 解码字符集,解析响应数据时用之,其为null时默认采用UTF-8解码
061     * @return 远程主机响应正文
062     */
063    public static String sendGetRequest(String reqURL, String decodeCharset){
064        long responseLength = 0;       //响应长度
065        String responseContent = null; //响应内容
066        HttpClient httpClient = new DefaultHttpClient(); //创建默认的httpClient实例
067        HttpGet httpGet = new HttpGet(reqURL);           //创建org.apache.http.client.methods.HttpGet
068        try{
069            HttpResponse response = httpClient.execute(httpGet); //执行GET请求
070            HttpEntity entity = response.getEntity();            //获取响应实体
071            if(null != entity){
072                responseLength = entity.getContentLength();
073                responseContent = EntityUtils.toString(entity, decodeCharset==null ? "UTF-8" : decodeCharset);
074                EntityUtils.consume(entity); //Consume response content
075            }
076            System.out.println("请求地址: " + httpGet.getURI());
077            System.out.println("响应状态: " + response.getStatusLine());
078            System.out.println("响应长度: " + responseLength);
079            System.out.println("响应内容: " + responseContent);
080        }catch(ClientProtocolException e){
081            LogUtil.getLogger().error("该异常通常是协议错误导致,比如构造HttpGet对象时传入的协议不对(将'http'写成'htp')或者服务器端返回的内容不符合HTTP协议要求等,堆栈信息如下", e);
082        }catch(ParseException e){
083            LogUtil.getLogger().error(e.getMessage(), e);
084        }catch(IOException e){
085            LogUtil.getLogger().error("该异常通常是网络原因引起的,如HTTP服务器未启动等,堆栈信息如下", e);
086        }finally{
087            httpClient.getConnectionManager().shutdown(); //关闭连接,释放资源
088        }
089        return responseContent;
090    }

116    public static String sendPostRequest(String reqURL, String sendData, boolean isEncoder, String encodeCharset, String decodeCharset){
117        String responseContent = null;
118        HttpClient httpClient = new DefaultHttpClient();
119         
120        HttpPost httpPost = new HttpPost(reqURL);
121        //httpPost.setHeader(HTTP.CONTENT_TYPE, "application/x-www-form-urlencoded; charset=UTF-8");
122        httpPost.setHeader(HTTP.CONTENT_TYPE, "application/x-www-form-urlencoded");
123        try{
124            if(isEncoder){
125                List<NameValuePair> formParams = new ArrayList<NameValuePair>();
126                for(String str : sendData.split("&")){
127                    formParams.add(new BasicNameValuePair(str.substring(0,str.indexOf("=")), str.substring(str.indexOf("=")+1)));
128                }
129                httpPost.setEntity(new StringEntity(URLEncodedUtils.format(formParams, encodeCharset==null ? "UTF-8" : encodeCharset)));
130            }else{
131                httpPost.setEntity(new StringEntity(sendData));
132            }
133             
134            HttpResponse response = httpClient.execute(httpPost);
135            HttpEntity entity = response.getEntity();
136            if (null != entity) {
137                responseContent = EntityUtils.toString(entity, decodeCharset==null ? "UTF-8" : decodeCharset);
138                EntityUtils.consume(entity);
139            }
140        }catch(Exception e){
141            LogUtil.getLogger().error("与[" + reqURL + "]通信过程中发生异常,堆栈信息如下", e);
142        }finally{
143            httpClient.getConnectionManager().shutdown();
144        }
145        return responseContent;
146    }

207    public static String sendPostSSLRequest(String reqURL, Map<String, String> params, String encodeCharset, String decodeCharset){
208        String responseContent = "";
209        HttpClient httpClient = new DefaultHttpClient();
210        X509TrustManager xtm = new X509TrustManager(){
211            public void checkClientTrusted(X509Certificate[] chain, String authType) throws CertificateException {}
212            public void checkServerTrusted(X509Certificate[] chain, String authType) throws CertificateException {}
213            public X509Certificate[] getAcceptedIssuers() {return null;}
214        };
215        try {
216            SSLContext ctx = SSLContext.getInstance("TLS");
217            ctx.init(null, new TrustManager[]{xtm}, null);
218            SSLSocketFactory socketFactory = new SSLSocketFactory(ctx);
219            httpClient.getConnectionManager().getSchemeRegistry().register(new Scheme("https", 443, socketFactory));
220             
221            HttpPost httpPost = new HttpPost(reqURL);
222            List<NameValuePair> formParams = new ArrayList<NameValuePair>();
223            for(Map.Entry<String,String> entry : params.entrySet()){
224                formParams.add(new BasicNameValuePair(entry.getKey(), entry.getValue()));
225            }
226            httpPost.setEntity(new UrlEncodedFormEntity(formParams, encodeCharset==null ? "UTF-8" : encodeCharset));
227             
228            HttpResponse response = httpClient.execute(httpPost);
229            HttpEntity entity = response.getEntity();
230            if (null != entity) {
231                responseContent = EntityUtils.toString(entity, decodeCharset==null ? "UTF-8" : decodeCharset);
232                EntityUtils.consume(entity);
233            }
234        } catch (Exception e) {
235            LogUtil.getLogger().error("与[" + reqURL + "]通信过程中发生异常,堆栈信息为", e);
236        } finally {
237            httpClient.getConnectionManager().shutdown();
238        }
239        return responseContent;

266     * @see 本方法默认的连接超时时间为30秒,默认的读取超时时间为30秒
267     * @param reqURL   请求地址
268     * @param sendData 发送到远程主机的正文数据
269     * @return 远程主机响应正文`HTTP状态码,如<code>"SUCCESS`200"</code><br>若通信过程中发生异常则返回"Failed`HTTP状态码",如<code>"Failed`500"</code>
270     */
271    public static String sendPostRequestByJava(String reqURL, String sendData){
272        HttpURLConnection httpURLConnection = null;
273        OutputStream out = null; //写
274        InputStream in = null;   //读
275        int httpStatusCode = 0;  //远程主机响应的HTTP状态码
276        try{
277            URL sendUrl = new URL(reqURL);
278            httpURLConnection = (HttpURLConnection)sendUrl.openConnection();
279            httpURLConnection.setRequestMethod("POST");
280            httpURLConnection.setDoOutput(true);        //指示应用程序要将数据写入URL连接,其值默认为false
281            httpURLConnection.setUseCaches(false);
282            httpURLConnection.setConnectTimeout(30000); //30秒连接超时
283            httpURLConnection.setReadTimeout(30000);    //30秒读取超时
284             
285            out = httpURLConnection.getOutputStream();
286            out.write(sendData.toString().getBytes());
287             
288            //清空缓冲区,发送数据
289            out.flush();
290             
291            //获取HTTP状态码
292            httpStatusCode = httpURLConnection.getResponseCode();
293             
294            //该方法只能获取到[HTTP/1.0 200 OK]中的[OK]
295            //若对方响应的正文放在了返回报文的最后一行,则该方法获取不到正文,而只能获取到[OK],稍显遗憾
296            //respData = httpURLConnection.getResponseMessage();
297             
298//          //处理返回结果
299//          BufferedReader br = new BufferedReader(new InputStreamReader(httpURLConnection.getInputStream()));
300//          String row = null;
301//          String respData = "";
302//          if((row=br.readLine()) != null){ //readLine()方法在读到换行[\n]或回车[\r]时,即认为该行已终止
303//              respData = row;              //HTTP协议POST方式的最后一行数据为正文数据
304//          }
305//          br.close();
306             
307            in = httpURLConnection.getInputStream();            
308            byte[] byteDatas = new byte[in.available()];
309            in.read(byteDatas);
310            return new String(byteDatas) + "`" + httpStatusCode;
311        }catch(Exception e){
312            LogUtil.getLogger().error(e.getMessage());
313            return "Failed`" + httpStatusCode;
314        }finally{
315            if(out != null){
316                try{
317                    out.close();
318                }catch (Exception e){
319                    LogUtil.getLogger().error("关闭输出流时发生异常,堆栈信息如下", e);
320                }
321            }
322            if(in != null){
323                try{
324                    in.close();
325                }catch(Exception e){
326                    LogUtil.getLogger().error("关闭输入流时发生异常,堆栈信息如下", e);
327                }
328            }
329            if(httpURLConnection != null){
330                httpURLConnection.disconnect();
331                httpURLConnection = null;
332            }
333        }
334    }
